32 TEST(MockSpannerClient, SuccessfulExecuteQuery) {
33  // Create a mock object to stream the results of a ExecuteQuery.
35  auto source =
36  std::unique_ptr<google::cloud::spanner_mocks::MockResultSetSource>(
37  new google::cloud::spanner_mocks::MockResultSetSource);
39 
40  // Setup the return type of the ExecuteQuery results:
42  google::spanner::v1::ResultSetMetadata metadata;
43  ASSERT_TRUE(google::protobuf::TextFormat::ParseFromString(
44  R"pb(
45  row_type: {
46  fields: {
47  name: "Id",
48  type: { code: INT64 }
49  }
50  fields: {
51  name: "Greeting",
52  type: { code: STRING }
53  }
54  }
55  )pb",
56  &metadata));
57  EXPECT_CALL(*source, Metadata()).WillRepeatedly(Return(metadata));
59 
60  // Setup the mock source to return some values:
62  EXPECT_CALL(*source, NextRow())
63  .WillOnce(Return(
64  spanner::MakeTestRow({{"Id", spanner::Value(1)},
65  {"Greeting", spanner::Value("Hello World")}})))
66  .WillOnce(Return(
67  spanner::MakeTestRow({{"Id", spanner::Value(2)},
68  {"Greeting", spanner::Value("Hello World")}})))
71  .WillOnce(Return(spanner::Row()));
73 
74  // Create a mock for `spanner::Connection`:
76  auto conn = std::make_shared<google::cloud::spanner_mocks::MockConnection>();
78 
79  // Setup the connection mock to return the results previously setup:
81  EXPECT_CALL(*conn, ExecuteQuery(_))
82  .WillOnce([&source](spanner::Connection::SqlParams const&)
83  -> spanner::RowStream {
84  return spanner::RowStream(std::move(source));
85  });

98 
100  int count = 0;
