66 StatusOr<google::iam::v1::Policy> DatabaseAdminClient::SetIamPolicy(
67  Database const& db, IamUpdater const& updater) {

69  auto default_rerun_policy =

72  auto const backoff_initial_delay = std::chrono::milliseconds(1000);

84 StatusOr<google::iam::v1::Policy> DatabaseAdminClient::SetIamPolicy(
85  Database const& db, IamUpdater const& updater,
86  std::unique_ptr<TransactionRerunPolicy> rerun_policy,
87  std::unique_ptr<BackoffPolicy> backoff_policy) {
88  using RerunnablePolicy = internal::SafeTransactionRerun;
89 
90  Status last_status;
91  do {
92  auto current_policy = GetIamPolicy(db);
93  if (!current_policy) {
94  last_status = std::move(current_policy).status();
95  } else {
96  auto etag = current_policy->etag();
97  auto desired = updater(*current_policy);
98  if (!desired.has_value()) {
99  return current_policy;
100  }
101  desired->set_etag(std::move(etag));
102  auto result = SetIamPolicy(db, *std::move(desired));
103  if (RerunnablePolicy::IsOk(result.status())) {
104  return result;
105  }
106  last_status = std::move(result).status();
107  }
108  if (!rerun_policy->OnFailure(last_status)) break;
109  std::this_thread::sleep_for(backoff_policy->OnCompletion());
110  } while (!rerun_policy->IsExhausted());
111  return last_status;
112 }
