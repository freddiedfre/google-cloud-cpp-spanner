159  ListDatabaseRange ListDatabases(ListDatabasesParams p) override {
160  gcsa::ListDatabasesRequest request;
161  request.set_parent(p.instance.FullName());
162  request.clear_page_token();
163  auto stub = stub_;
164  // Because we do not have C++14 generalized lambda captures we cannot just
165  // use the unique_ptr<> here, so convert to shared_ptr<> instead.
166  auto retry =
167  std::shared_ptr<RetryPolicy const>(retry_policy_prototype_->clone());
168  auto backoff = std::shared_ptr<BackoffPolicy const>(
169  backoff_policy_prototype_->clone());
170 
171  char const* function_name = __func__;
172  return ListDatabaseRange(
173  std::move(request),
174  [stub, retry, backoff,
175  function_name](gcsa::ListDatabasesRequest const& r) {
176  return RetryLoop(
177  retry->clone(), backoff->clone(), true,
178  [stub](grpc::ClientContext& context,
179  gcsa::ListDatabasesRequest const& request) {
180  return stub->ListDatabases(context, request);
181  },
182  r, function_name);
183  },
184  [](gcsa::ListDatabasesResponse r) {
185  std::vector<gcsa::Database> result(r.databases().size());
186  auto& dbs = *r.mutable_databases();
187  std::move(dbs.begin(), dbs.end(), result.begin());
188  return result;
