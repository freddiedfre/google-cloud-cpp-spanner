48 TracingOptions& TracingOptions::SetOptions(std::string const& str) {
49  auto const beg = str.begin();
50  auto const end = str.end();
51  for (auto pos = beg;;) {
52  auto const comma = std::find(pos, end, ',');
53  auto const equal = std::find(pos, comma, '=');
54  std::string const opt{pos, equal};
55  std::string const val{equal + (equal == comma ? 0 : 1), comma};
56  if (opt == "single_line_mode") {
57  if (auto v = ParseBoolean(val)) single_line_mode_ = *v;
58  } else if (opt == "use_short_repeated_primitives") {
59  if (auto v = ParseBoolean(val)) use_short_repeated_primitives_ = *v;
60  } else if (opt == "truncate_string_field_longer_than") {
61  if (auto v = ParseInteger(val)) truncate_string_field_longer_than_ = *v;
62  }
63  if (comma == end) break;
64  pos = comma + 1;
65  }
66  return *this;
67 }
