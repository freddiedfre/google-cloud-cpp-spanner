35 // Override connection endpoint and credentials with values appropriate for
36 // an emulated backend. This should be done after any user code that could
37 // also override the default values (i.e., immediately before establishing
38 // the connection).
39 ConnectionOptions EmulatorOverrides(ConnectionOptions options) {
40  auto emulator_addr = google::cloud::internal::GetEnv("SPANNER_EMULATOR_HOST");
41  if (emulator_addr.has_value()) {
42  options.set_endpoint(*emulator_addr)
43  .set_credentials(grpc::InsecureChannelCredentials());
44  }
45  return options;
46 }

85 grpc::ChannelArguments ConnectionOptions::CreateChannelArguments() const {
86  grpc::ChannelArguments channel_arguments;
87 
88  if (!channel_pool_domain().empty()) {
89  // To get a different channel pool one just needs to set any channel
90  // parameter to a different value. Newer versions of gRPC include a macro
91  // for this purpose (GRPC_ARG_CHANNEL_POOL_DOMAIN). As we are compiling
92  // against older versions in some cases, we use the actual value.
93  channel_arguments.SetString("grpc.channel_pooling_domain",
94  channel_pool_domain());
95  }
96  channel_arguments.SetUserAgentPrefix(user_agent_prefix());
97  return channel_arguments;
98 }
